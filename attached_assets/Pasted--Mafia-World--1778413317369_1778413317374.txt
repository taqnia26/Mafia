// GET /api/nuclear/status - حالة المفاعل النووي
router.get('/status', requireAuth, async (req, res) => {
  const player = await db.select().from(players)
    .where(eq(players.clerkId, req.auth.userId))
    .limit(1);

  if (!player[0]) {
    return res.status(404).json({ error: 'Player not found' });
  }

  // جلب المفاعل
  const reactorType = await db.select().from(propertyTypes)
    .where(eq(propertyTypes.nameEn, 'Nuclear Reactor'))
    .limit(1);

  const reactor = await db.select({
    id: playerProperties.id,
    level: playerProperties.level,
    isUnderConstruction: playerProperties.isUnderConstruction,
    constructionCompleteAt: playerProperties.constructionCompleteAt,
    disabledUntil: playerProperties.disabledUntil,
    cityId: playerProperties.cityId
  })
  .from(playerProperties)
  .where(and(
    eq(playerProperties.playerId, player[0].id),
    eq(playerProperties.propertyTypeId, reactorType[0].id)
  ))
  .limit(1);

  if (!reactor[0]) {
    return res.json({ hasReactor: false });
  }

  // جلب الطاقة
  const energy = await db.select().from(nuclearEnergy)
    .where(and(
      eq(nuclearEnergy.playerId, player[0].id),
      eq(nuclearEnergy.reactorPropertyId, reactor[0].id)
    ))
    .limit(1);

  // التحقق من الدرع النشط
  const shield = await db.select().from(nuclearShields)
    .where(and(
      eq(nuclearShields.playerId, player[0].id),
      gt(nuclearShields.expiresAt, new Date())
    ))
    .limit(1);

  res.json({
    hasReactor: true,
    reactor: reactor[0],
    energy: energy[0]?.units || 0,
    lastGenerated: energy[0]?.lastGeneratedAt,
    shield: shield[0] || null,
    canStrike: player[0].lastNukeStrikeAt 
      ? (Date.now() - new Date(player[0].lastNukeStrikeAt).getTime()) > 7 * 24 * 60 * 60 * 1000
      : true,
    canShield: player[0].lastShieldActivationAt
      ? (Date.now() - new Date(player[0].lastShieldActivationAt).getTime()) > 14 * 24 * 60 * 60 * 1000
      : true
  });
});

router.post('/purchase', requireAuth, async (req, res) => {
  const { cityId } = req.body;

  await db.transaction(async (tx) => {
    const player = await tx.select().from(players)
      .where(eq(players.clerkId, req.auth.userId))
      .limit(1);

    // التحقق من الرتبة (11 = ناشونال باشا)
    const currentRank = await tx.select().from(playerRanks)
      .where(eq(playerRanks.rankNumber, player[0].currentRankNumber || 1))
      .limit(1);

    if ((player[0].currentRankNumber || 1) < 11) {
      throw new Error('Must be National Pasha (Rank 11) to purchase Nuclear Reactor');
    }

    // التحقق من الحد الأقصى في المدينة
    const cityLimit = await tx.select().from(nuclearReactorLimits)
      .where(eq(nuclearReactorLimits.cityId, cityId))
      .limit(1);

    if (cityLimit[0] && cityLimit[0].reactorCount >= 3) {
      throw new Error('Maximum 3 nuclear reactors per city reached');
    }

    // التحقق من المال
    const reactorType = await tx.select().from(propertyTypes)
      .where(eq(propertyTypes.nameEn, 'Nuclear Reactor'))
      .limit(1);

    if (player[0].money < reactorType[0].price) {
      throw new Error('Insufficient money');
    }

    // خصم المال
    await tx.update(players)
      .set({ 
        money: sql`${players.money} - ${reactorType[0].price}`,
        hasNuclearReactor: true
      })
      .where(eq(players.id, player[0].id));

    // إنشاء المفاعل (تحت الإنشاء 7 أيام)
    const constructionComplete = new Date(Date.now() + 7 * 24 * 60 * 60 * 1000);

    const [newReactor] = await tx.insert(playerProperties).values({
      playerId: player[0].id,
      propertyTypeId: reactorType[0].id,
      level: 1,
      cityId,
      isUnderConstruction: true,
      constructionCompleteAt: constructionComplete,
      lastIncomeCollectedAt: new Date()
    }).returning();

    // إنشاء سجل الطاقة
    await tx.insert(nuclearEnergy).values({
      playerId: player[0].id,
      reactorPropertyId: newReactor.id,
      units: 0,
      lastGeneratedAt: constructionComplete // يبدأ الإنتاج بعد البناء
    });

    // تحديث عداد المدينة
    await tx.insert(nuclearReactorLimits).values({
      cityId,
      reactorCount: 1
    }).onConflictDoUpdate({
      target: nuclearReactorLimits.cityId,
      set: { 
        reactorCount: sql`${nuclearReactorLimits.reactorCount} + 1`,
        lastUpdatedAt: new Date()
      }
    });

    // منح سلاح القنبلة القذرة
    const dirtyBomb = await tx.select().from(weapons)
      .where(eq(weapons.type, 'nuclear'))
      .limit(1);

    await tx.insert(playerWeapons).values({
      playerId: player[0].id,
      weaponId: dirtyBomb[0].id,
      quantity: 1
    }).onConflictDoUpdate({
      target: [playerWeapons.playerId, playerWeapons.weaponId],
      set: { quantity: sql`${playerWeapons.quantity} + 1` }
    });

    res.json({
      success: true,
      reactor: newReactor,
      constructionComplete
    });
  });
});

router.post('/strike', requireAuth, async (req, res) => {
  const { targetPropertyId } = req.body;

  await db.transaction(async (tx) => {
    const player = await tx.select().from(players)
      .where(eq(players.clerkId, req.auth.userId))
      .limit(1);

    // التحقق من Cooldown (7 أيام)
    if (player[0].lastNukeStrikeAt) {
      const daysSince = (Date.now() - new Date(player[0].lastNukeStrikeAt).getTime()) / (1000 * 60 * 60 * 24);
      if (daysSince < 7) {
        throw new Error(`Cooldown active. ${Math.ceil(7 - daysSince)} days remaining`);
      }
    }

    // التحقق من الطاقة
    const energy = await tx.select().from(nuclearEnergy)
      .where(eq(nuclearEnergy.playerId, player[0].id))
      .limit(1);

    if (!energy[0] || energy[0].units < 1) {
      throw new Error('Insufficient nuclear energy (need 1 unit)');
    }

    // جلب العقار المستهدف
    const targetProperty = await tx.select({
      id: playerProperties.id,
      ownerId: playerProperties.playerId,
      cityId: playerProperties.cityId,
      disabledUntil: playerProperties.disabledUntil
    })
    .from(playerProperties)
    .where(eq(playerProperties.id, targetPropertyId))
    .limit(1);

    if (!targetProperty[0]) {
      throw new Error('Target property not found');
    }

    if (targetProperty[0].ownerId === player[0].id) {
      throw new Error('Cannot nuke your own property');
    }

    // التحقق من درع نووي نشط
    const targetShield = await tx.select().from(nuclearShields)
      .where(and(
        eq(nuclearShields.playerId, targetProperty[0].ownerId),
        gt(nuclearShields.expiresAt, new Date())
      ))
      .limit(1);

    if (targetShield[0]) {
      throw new Error('Target is protected by Nuclear Shield');
    }

    // خصم الطاقة
    await tx.update(nuclearEnergy)
      .set({ units: sql`${nuclearEnergy.units} - 1` })
      .where(eq(nuclearEnergy.id, energy[0].id));

    // تعطيل العقار لمدة 72 ساعة
    const disabledUntil = new Date(Date.now() + 72 * 60 * 60 * 1000);

    await tx.update(playerProperties)
      .set({ disabledUntil })
      .where(eq(playerProperties.id, targetPropertyId));

    // تسجيل الضربة
    await tx.insert(nuclearStrikes).values({
      attackerId: player[0].id,
      targetPropertyId,
      propertyOwnerId: targetProperty[0].ownerId,
      disabledUntil,
      energyUsed: 1
    });

    // تحديث Cooldown
    await tx.update(players)
      .set({ lastNukeStrikeAt: new Date() })
      .where(eq(players.id, player[0].id));

    // إشعارات
    await tx.insert(notifications).values({
      playerId: targetProperty[0].ownerId,
      type: 'nuclear_strike',
      message: `☢️ NUCLEAR STRIKE! ${player[0].username} hit your property. Disabled for 72 hours!`,
      messageAr: `☢️ ضربة نووية! ${player[0].username} دمر عقارك. معطل لمدة 72 ساعة!`,
      link: '/properties'
    });

    res.json({ success: true, disabledUntil });
  });
});

    const energy = await tx.select().from(nuclearEnergy)
      .where(eq(nuclearEnergy.playerId, player[0].id))
      .limit(1);

    if (!energy[0] || energy[0].units < 2) {
      throw new Error('Insufficient nuclear energy (need 2 units)');
    }

    // خصم الطاقة
    await tx.update(nuclearEnergy)
      .set({ units: sql`${nuclearEnergy.units} - 2` })
      .where(eq(nuclearEnergy.id, energy[0].id));

    // تفعيل الدرع
    const expiresAt = new Date(Date.now() + 48 * 60 * 60 * 1000);

    await tx.insert(nuclearShields).values({
      playerId: player[0].id,
      expiresAt,
      energyUsed: 2
    }).onConflictDoUpdate({
      target: nuclearShields.playerId,
      set: { 
        activatedAt: new Date(),
        expiresAt,
        energyUsed: 2
      }
    });

    await tx.update(players)
      .set({ lastShieldActivationAt: new Date() })
      .where(eq(players.id, player[0].id));

    res.json({ success: true, expiresAt });
  });
});

async function generateNuclearEnergy() {
  logger.info('Generating nuclear energy...');

  await db.transaction(async (tx) => {
    // جلب كل المفاعلات النشطة (غير قيد الإنشاء)
    const reactors = await tx.select({
      energyId: nuclearEnergy.id,
      units: nuclearEnergy.units,
      lastGenerated: nuclearEnergy.lastGeneratedAt,
      reactorId: playerProperties.id,
      isConstruction: playerProperties.isUnderConstruction,
      constructionComplete: playerProperties.constructionCompleteAt
    })
    .from(nuclearEnergy)
    .innerJoin(playerProperties, eq(nuclearEnergy.reactorPropertyId, playerProperties.id))
    .where(eq(playerProperties.isUnderConstruction, false));

    const now = new Date();

    for (const reactor of reactors) {
      const hoursSinceGeneration = (now.getTime() - new Date(reactor.lastGenerated).getTime()) / (1000 * 60 * 60);

      // إنتاج وحدة كل 24 ساعة
      if (hoursSinceGeneration >= 24) {
        const unitsToGenerate = Math.floor(hoursSinceGeneration / 24);

        await tx.update(nuclearEnergy)
          .set({
            units: sql`${nuclearEnergy.units} + ${unitsToGenerate}`,
            lastGeneratedAt: now
          })
          .where(eq(nuclearEnergy.id, reactor.energyId));

        logger.info({ reactorId: reactor.reactorId, generated: unitsToGenerate }, 'Nuclear energy generated');
      }
    }

    // إكمال بناء المفاعلات
    await tx.update(playerProperties)
      .set({ isUnderConstruction: false })
      .where(and(
        eq(playerProperties.isUnderConstruction, true),
        sql`${playerProperties.constructionCompleteAt} <= NOW()`
      ));
  });
}

router.get('/status', requireAuth, async (req, res) => {

  // القروض النشطة
  const loans = await db.select().from(bankLoans)
    .where(and(
      eq(bankLoans.borrowerId, player[0].id),
      eq(bankLoans.status, 'active')
    ));

  res.json({
    deposits,
    loans,
    isBlacklisted: player[0].isBankBlacklisted,
    totalDeposited: player[0].totalBankDeposits,
    totalBorrowed: player[0].totalBankLoans
  });
});

// POST /api/bank/deposit - إيداع
router.post('/deposit', requireAuth, async (req, res) => {
  const { amount, durationDays, bankPropertyId } = req.body;

  const rates = {
    7: 0.006,   // 0.6%
    14: 0.010,  // 1.0%
    30: 0.015   // 1.5%
  };

  if (!rates[durationDays as keyof typeof rates]) {
    return res.status(400).json({ error: 'Invalid duration' });
  }

  if (amount < 100000 || amount > 100000000) {
    return res.status(400).json({ error: 'Amount must be between $100k - $100M' });
  }

  await db.transaction(async (tx) => {
    const player = await tx.select().from(players)
      .where(eq(players.clerkId, req.auth.userId))
      .limit(1);

    if (player[0].isBankBlacklisted) {
      throw new Error('You are blacklisted from banks');
    }

    if (player[0].money < amount) {
      throw new Error('Insufficient money');
    }

    // جلب البنك
    const bank = await tx.select().from(playerProperties)
      .where(eq(playerProperties.id, bankPropertyId))
      .limit(1);

    if (!bank[0]) {
      throw new Error('Bank not found');
    }

    // رسوم الإيداع 2% للبنك
    const depositFee = Math.floor(amount * 0.02);
    const netDeposit = amount - depositFee;

    // خصم المال
    await tx.update(players)
      .set({ 
        money: sql`${players.money} - ${amount}`,
        totalBankDeposits: sql`${players.totalBankDeposits} + ${netDeposit}`
      })
      .where(eq(players.id, player[0].id));

    // إضافة رسوم للبنك
    await tx.update(players)
      .set({ money: sql`${players.money} + ${depositFee}` })
      .where(eq(players.id, bank[0].playerId));

    // إنشاء الإيداع
    const maturesAt = new Date(Date.now() + durationDays * 24 * 60 * 60 * 1000);

    const [deposit] = await tx.insert(bankDeposits).values({
      playerId: player[0].id,
      bankPropertyId,
      bankOwnerId: bank[0].playerId,
      amount: netDeposit,
      durationDays,
      dailyInterestRate: rates[durationDays as keyof typeof rates].toString(),
      maturesAt
    }).returning();

    res.json({ success: true, deposit, fee: depositFee });
  });
});

router.post('/withdraw/:depositId', requireAuth, async (req, res) => {
  const { depositId } = req.params;
  const { early } = req.body; // هل سحب مبكر؟

  await db.transaction(async (tx) => {
    const player = await tx.select().from(players)
      .where(eq(players.clerkId, req.auth.userId))
      .limit(1);

    const deposit = await tx.select().from(bankDeposits)
      .where(and(
        eq(bankDeposits.id, Number(depositId)),
        eq(bankDeposits.playerId, player[0].id),
        eq(bankDeposits.status, 'active')
      ))
      .limit(1);

    if (!deposit[0]) {
      throw new Error('Deposit not found');
    }

    const now = new Date();
    const isMatured = now >= new Date(deposit[0].maturesAt);

    let finalAmount: number;
    let bankOwnerGain = 0;

    if (early || !isMatured) {
      // سحب مبكر: خسارة الفوائد + رسوم 5%
      const penalty = Math.floor(deposit[0].amount * 0.05);
      finalAmount = deposit[0].amount - penalty;
      bankOwnerGain = Math.floor(penalty * 0.5); // 50% من الرسوم للبنك

      await tx.update(bankDeposits)
        .set({
          status: 'early_withdrawn',
          withdrawnAt: now,
          finalAmount
        })
        .where(eq(bankDeposits.id, deposit[0].id));

    } else {
      // سحب عادي: حساب الفوائد
      const days = deposit[0].durationDays;
      const rate = parseFloat(deposit[0].dailyInterestRate);
      const totalInterest = Math.floor(deposit[0].amount * rate * days);
      finalAmount = deposit[0].amount + totalInterest;

      await tx.update(bankDeposits)
        .set({
          status: 'withdrawn',
          withdrawnAt: now,
          finalAmount
        })
        .where(eq(bankDeposits.id, deposit[0].id));
    }

    // إضافة المال للاعب
    await tx.update(players)
      .set({
        money: sql`${players.money} + ${finalAmount}`,
        totalBankDeposits: sql`${players.totalBankDeposits} - ${deposit[0].amount}`
      })
      .where(eq(players.id, player[0].id));

    // مكافأة صاحب البنك
    if (bankOwnerGain > 0) {
      await tx.update(players)
        .set({ money: sql`${players.money} + ${bankOwnerGain}` })
        .where(eq(players.id, deposit[0].bankOwnerId));
    }

    res.json({ success: true, finalAmount, penalty: early });
  });
});

        eq(bankLoans.borrowerId, player[0].id),
        eq(bankLoans.status, 'active')
      ))
      .limit(1);

    if (existingLoan[0]) {
      throw new Error('You already have an active loan');
    }

    const bank = await tx.select().from(playerProperties)
      .where(eq(playerProperties.id, bankPropertyId))
      .limit(1);

    const interestRate = 0.25; // 25%
    const totalAmount = Math.floor(amount * (1 + interestRate));
    const dueAt = new Date(Date.now() + 14 * 24 * 60 * 60 * 1000); // 14 يوم

    // إعطاء المال للاعب
    await tx.update(players)
      .set({
        money: sql`${players.money} + ${amount}`,
        totalBankLoans: sql`${players.totalBankLoans} + ${totalAmount}`
      })
      .where(eq(players.id, player[0].id));

    // إنشاء القرض
    const [loan] = await tx.insert(bankLoans).values({
      borrowerId: player[0].id,
      bankPropertyId,
      bankOwnerId: bank[0].playerId,
      principalAmount: amount,
      interestRate: interestRate.toString(),
      totalAmount,
      dueAt
    }).returning();

    res.json({ success: true, loan, totalDue: totalAmount });
  });
});

router.post('/repay/:loanId', requireAuth, async (req, res) => {
  const { loanId } = req.params;

  await db.transaction(async (tx) => {
    const player = await tx.select().from(players)
      .where(eq(players.clerkId, req.auth.userId))
      .limit(1);

    const loan = await tx.select().from(bankLoans)
      .where(and(
        eq(bankLoans.id, Number(loanId)),
        eq(bankLoans.borrowerId, player[0].id),
        eq(bankLoans.status, 'active')
      ))
      .limit(1);

    if (!loan[0]) {
      throw new Error('Loan not found');
    }

    if (player[0].money < loan[0].totalAmount) {
      throw new Error('Insufficient money to repay');
    }

    // خصم المال
    await tx.update(players)
      .set({
        money: sql`${players.money} - ${loan[0].totalAmount}`,
        totalBankLoans: sql`${players.totalBankLoans} - ${loan[0].totalAmount}`
      })
      .where(eq(players.id, player[0].id));

    // إضافة للبنك (المبلغ الكامل بالفائدة)
    await tx.update(players)
      .set({ money: sql`${players.money} + ${loan[0].totalAmount}` })
      .where(eq(players.id, loan[0].bankOwnerId));

    // تحديث القرض
    await tx.update(bankLoans)
      .set({
        status: 'repaid',
        repaidAt: new Date()
      })
      .where(eq(bankLoans.id, loan[0].id));

    res.json({ success: true });
  });
});

// أضف دالة للتحقق من القروض المتأخرة

async function checkOverdueLoans() {
  logger.info('Checking overdue loans...');

  await db.transaction(async (tx) => {
    const overdueLoans = await tx.select()
      .from(bankLoans)
      .where(and(
        eq(bankLoans.status, 'active'),
        sql`${bankLoans.dueAt} < NOW()`
      ));

    for (const loan of overdueLoans) {
      // وضع في القائمة السوداء
      await tx.insert(bankBlacklist).values({
        playerId: loan.borrowerId,
        reason: `Failed to repay loan of $${loan.totalAmount}`
      }).onConflictDoNothing();

      await tx.update(players)
        .set({ isBankBlacklisted: true })
        .where(eq(players.id, loan.borrowerId));

      // تحديث القرض
      await tx.update(bankLoans)
        .set({ status: 'defaulted' })
        .where(eq(bankLoans.id, loan.id));

      // إشعار
      await tx.insert(notifications).values({
        playerId: loan.borrowerId,
        type: 'loan_default',
        message: `⚠️ LOAN DEFAULT! You are now blacklisted from all banks.`,
        messageAr: `⚠️ تعثر في السداد! أنت الآن في القائمة السوداء لكل البنوك.`,
        link: '/bank'
      });

      logger.warn({ loanId: loan.id, borrowerId: loan.borrowerId }, 'Loan defaulted');
    }
  });
}

// استدعاء كل ساعة
setInterval(checkOverdueLoans, 60 * 60 * 1000);
