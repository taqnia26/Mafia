## 2.3 Build Property Logic

```typescript
// lib/properties/canBuildProperty.ts
import { db } from '@/db';
import { users, propertyLimits, cityPropertyCounts, properties } from '@/db/schema';
import { eq, and } from 'drizzle-orm';

export async function canBuildProperty(
  userId: string,
  cityId: number,
  propertyType: string
) {
  // Get user
  const [user] = await db
    .select()
    .from(users)
    .where(eq(users.id, userId));
  
  if (!user) {
    return { canBuild: false, reason: 'User not found' };
  }
  
  // Get property limits
  const [limit] = await db
    .select()
    .from(propertyLimits)
    .where(eq(propertyLimits.propertyType, propertyType));
  
  if (!limit) {
    return { canBuild: false, reason: 'Invalid property type' };
  }
  
  // Check rank requirement
  if (limit.minRank && user.rank < limit.minRank) {
    return {
      canBuild: false,
      reason: `You need rank ${limit.minRank}+ to build this property`
    };
  }
  
  // Check money
  if (limit.priceCash && user.money < limit.priceCash) {
    return {
      canBuild: false,
      reason: `Not enough money. Need $${limit.priceCash.toLocaleString()}`
    };
  }
  
  // Check city limits (if limited)
  if (limit.maxPerCity !== -1) {
    const [cityCount] = await db
      .select()
      .from(cityPropertyCounts)
      .where(
        and(
          eq(cityPropertyCounts.cityId, cityId),
          eq(cityPropertyCounts.propertyType, propertyType)
        )
      );
    
    if (cityCount && cityCount.currentCount >= cityCount.maxCount) {
      // Get current owners
      const currentOwners = await db
        .select({
          ownerId: properties.ownerId,
          username: users.username
        })
        .from(properties)
        .innerJoin(users, eq(properties.ownerId, users.id))
        .where(
          and(
            eq(properties.cityId, cityId),
            eq(properties.propertyType, propertyType)
          )
        );
      
      return {
        canBuild: false,
        reason: `Maximum ${limit.maxPerCity} ${propertyType} per city. All slots taken!`,
        currentOwners
      };
    }
  }
  
  return { canBuild: true };
}

// Build property
export async function buildProperty(
  userId: string,
  cityId: number,
  propertyType: string,
  name: string
) {
  const check = await canBuildProperty(userId, cityId, propertyType);
  
  if (!check.canBuild) {
    throw new Error(check.reason);
  }
  
  const [limit] = await db
    .select()
    .from(propertyLimits)
    .where(eq(propertyLimits.propertyType, propertyType));
  
  // Deduct money
  await db
    .update(users)
    .set({ money: sql`money - ${limit.priceCash}` })
    .where(eq(users.id, userId));
  
  // Create property
  const [property] = await db
    .insert(properties)
    .values({
      ownerId: userId,
      cityId,
      propertyType,
      name,
      incomePerHour: limit.baseIncomePerHour
    })
    .returning();
  
  // Update city count (if limited)
  if (limit.maxPerCity !== -1) {
    await db
      .update(cityPropertyCounts)
      .set({ currentCount: sql`current_count + 1` })
      .where(
        and(
          eq(cityPropertyCounts.cityId, cityId),
          eq(cityPropertyCounts.propertyType, propertyType)
        )
      );
  }
  
  return property;
}
```

## 3.2 Rental Logic (WITHOUT user ID 4000)

```typescript
// lib/safeHouse/rentSafeHouse.ts
import { db } from '@/db';
import { users, properties, safeHouseRentals } from '@/db/schema';
import { eq, and, gt, sql } from 'drizzle-orm';

const OWNER_CUT = 0.35; // 35% to owner (better for sustainability)
const ADMIN_CUT = 0.65; // 65% to admin wallet (money sink)

export async function rentSafeHouse(
  renterId: string,
  propertyId: number,
  rentAmount: number,
  durationDays: number = 7 // Max 7 days
) {
  // Validations
  if (rentAmount < 10000) {
    throw new Error('Minimum rent is $10,000');
  }
  
  if (durationDays > 7) {
    throw new Error('Maximum rental period is 7 days');
  }
  
  // Get renter
  const [renter] = await db
    .select()
    .from(users)
    .where(eq(users.id, renterId));
  
  if (!renter) {
    throw new Error('Renter not found');
  }
  
  // Get property
  const [property] = await db
    .select()
    .from(properties)
    .where(eq(properties.id, propertyId));
  
  if (!property) {
    throw new Error('Property not found');
  }
  
  // Check same city
  if (property.cityId !== renter.cityId) {
    throw new Error('Safe House must be in your current city');
  }
  
  // Check cooldown (24 hours)
  const [lastRental] = await db
    .select()
    .from(safeHouseRentals)
    .where(
      and(
        eq(safeHouseRentals.renterId, renterId),
        gt(safeHouseRentals.createdAt, sql`NOW() - INTERVAL '24 hours'`)
      )
    );
  
  if (lastRental) {
    throw new Error('You can only rent once every 24 hours');
  }
  
  // Check money
  if (renter.money < rentAmount) {
    throw new Error('Not enough money');
  }
  
  // Calculate split
  const ownerRevenue = Math.floor(rentAmount * OWNER_CUT);
  const adminRevenue = rentAmount - ownerRevenue;
  
  const endTime = new Date();
  endTime.setDate(endTime.getDate() + durationDays);
  
  // Transaction
  await db.transaction(async (tx) => {
    // Deduct from renter
    await tx
      .update(users)
      .set({ money: sql`money - ${rentAmount}` })
      .where(eq(users.id, renterId));
    
    // Pay owner
    await tx
      .update(users)
      .set({ money: sql`money + ${ownerRevenue}` })
      .where(eq(users.id, property.ownerId));
    
    // Admin revenue goes to admin wallet (tracked separately, not to a user)
    // We'll add it to a separate admin_wallet table or track in adminCredentials
    
    // Create rental record
    await tx
      .insert(safeHouseRentals)
      .values({
        propertyId,
        renterId,
        ownerId: property.ownerId,
        rentAmount,
        ownerRevenue,
        adminRevenue,
        startTime: new Date(),
        endTime,
        status: 'active',
        renterIp: renter.ipAddress
      });
    
    // Activate safe house protection
    await tx
      .update(users)
      .set({
        inSafeHouse: true,
        safeHouseExpiresAt: endTime
      })
      .where(eq(users.id, renterId));
  });
  
  return {
    success: true,
    rentPaid: rentAmount,
    ownerReceived: ownerRevenue,
    adminReceived: adminRevenue,
    protectedUntil: endTime
  };
}

// Check if player can be attacked
export async function canAttackTarget(targetId: string) {

  if (target.inSafeHouse && target.expiresAt && target.expiresAt > new Date()) {

    };
  }
```

```typescript
      .select()
      .from(weapons)
      .where(eq(weapons.id, attacker.equippedWeaponId));
    
    if (!weapon) {
      return NextResponse.json({ error: 'No weapon equipped' }, { status: 400 });
    }
    
    // Get attacker rank bonus
    const [attackerRankData] = await db
      .select({ atkBonus: ranks.atkBonus })
      .from(ranks)
      .where(eq(ranks.id, attacker.rank));
    
    const attackerATK = weapon.atk + (attackerRankData?.atkBonus || 0);
    
    // Get target rank data
    const [targetRankData] = await db
      .select()
      .from(ranks)
      .where(eq(ranks.id, targetRank));
    
    if (!targetRankData) {
      return NextResponse.json({ error: 'Invalid target rank' }, { status: 400 });
    }
    
    // Get target armor
    let armorDEF = 0;
    if (targetArmor && targetArmor !== 'none') {
      const [armorData] = await db
        .select({ def: armor.def })
        .from(armor)
        .where(eq(armor.id, parseInt(targetArmor)));
      
      armorDEF = armorData?.def || 0;
    }
    
    const targetDEF = targetRankData.defBonus + armorDEF;
    const targetHP = targetRankData.maxHp;
    
    // Calculate damage
    const damagePerBullet = Math.max(attackerATK - targetDEF, 5);
    
    // Calculate bullets for guards
    let totalBulletsForGuards = 0;
    const guardDetails = [];
    
    for (const guard of guards) {
      const [guardData] = await db
        .select()
        .from(guardTypes)
        .where(eq(guardTypes.id, guard.typeId));
      
      if (guardData) {
        const guardDamage = Math.max(attackerATK - guardData.def, 5);
        const bulletsPerGuard = Math.ceil(guardData.hp / guardDamage);
        const totalForThisType = bulletsPerGuard * guard.count;
        
        totalBulletsForGuards += totalForThisType;
        
        guardDetails.push({
          type: guardData.nameEn,
          typeAr: guardData.nameAr,
          count: guard.count,
          bulletsPerGuard,
          totalBullets: totalForThisType
        });
      }
    }
    
    // Calculate bullets for target
    const bulletsForTarget = Math.ceil(targetHP / damagePerBullet);
    
    // Total
    const totalBullets = totalBulletsForGuards + bulletsForTarget;
    
    // Get bullet price
    const bulletPrice = 300; // TODO: Get from ammo_types table
    const totalCost = totalBullets * bulletPrice;
    
    // Check availability
    // TODO: Get actual ammo count from user_ammo
    const availableAmmo = 1000; // Placeholder
    const hasEnoughAmmo = availableAmmo >= totalBullets;
    const hasEnoughMoney = attacker.money >= totalCost;
    
    // Suggestions
    const suggestions = [];
    
    if (!hasEnoughAmmo) {
      suggestions.push({
        type: 'buy_ammo',
        message: `You need ${totalBullets - availableAmmo} more bullets`,
        cost: (totalBullets - availableAmmo) * bulletPrice
      });
    }
    
    // TODO: Suggest stronger weapons, boosts, etc.
    
    return NextResponse.json({
      attacker: {
        name: attacker.username,
        rank: attacker.rank,
        weapon: weapon.nameEn,
        totalATK: attackerATK
      },
      target: {
        rank: targetRank,
        armor: targetArmor,
        totalDEF: targetDEF,
        hp: targetHP
      },
      calculation: {
        damagePerBullet,
        guards: guardDetails,
        bulletsForGuards: totalBulletsForGuards,
        bulletsForTarget,
        totalBullets,
        bulletType: weapon.ammoType,
        totalCost
      },
      availability: {
        hasEnoughAmmo,
        hasEnoughMoney,
        availableAmmo,
        neededAmmo: Math.max(0, totalBullets - availableAmmo),
        neededMoney: Math.max(0, totalCost - attacker.money)
      },
      canAttack: hasEnoughAmmo && hasEnoughMoney,
      suggestions
    });
  } catch (error) {
    console.error('Kill calculator error:', error);
    return NextResponse.json(
      { error: 'Internal server error' },
      { status: 500 }
    );
```

## 5.2 Blackjack (Real Game, Not Fake Random)

**Why Blackjack First:**
- Most popular casino game
- Real skill-based decisions (Hit/Stand)
- Tests the casino commission system
- Foundation for other games later

**Game Rules:**
- Commission: 15% on every bet
- Min bet: $5,000
- Max bet: $50,000
- Payout: 1.8× (reduced from standard 2.0×)
- Tie = Dealer wins (house edge)
- No Double Down or Split (simplified)
- Daily limit: 20 games per player

```typescript
// lib/casino/blackjack.ts
import { db } from '@/db';
import { users, properties, casinoGames, adminWallet } from '@/db/schema';
import { eq, sql } from 'drizzle-orm';

const COMMISSION = 0.15; // 15%
const MIN_BET = 5000;
const MAX_BET = 50000;
const PAYOUT_MULTIPLIER = 1.8; // Reduced from 2.0

type Card = number; // 1-13 (Ace-King)
type Hand = Card[];

function dealCard(): Card {
  return Math.floor(Math.random() * 13) + 1;
}

function calculateHandValue(hand: Hand): number {
  let total = 0;
  let aces = 0;
  
  for (const card of hand) {
    if (card === 1) {
      aces++;
      total += 11;
    } else if (card > 10) {
      total += 10; // Face cards = 10
    } else {
      total += card;
    }
  }
  
  // Adjust aces if needed
  while (total > 21 && aces > 0) {
    total -= 10;
    aces--;
  }
  
  return total;
}

export async function playBlackjack(
  playerId: string,
  casinoId: number,
  betAmount: number,
  actions: ('hit' | 'stand')[] // Player decisions
) {
  // Validations
  if (betAmount < MIN_BET || betAmount > MAX_BET) {
    throw new Error(`Bet must be between $${MIN_BET} and $${MAX_BET}`);
  }
  
  // Get player
  const [player] = await db
    .select()
    .from(users)
    .where(eq(users.id, playerId));
  
  if (player.money < betAmount) {
    throw new Error('Not enough money');
  }
  
  // Get casino
  const [casino] = await db
    .select()
    .from(properties)
    .where(eq(properties.id, casinoId));
  
  // Calculate commission
  const commission = Math.floor(betAmount * COMMISSION);
  const effectiveBet = betAmount - commission;
  
  // Deal initial hands
  const playerHand: Hand = [dealCard(), dealCard()];
  const dealerHand: Hand = [dealCard(), dealCard()];
  
  let playerTotal = calculateHandValue(playerHand);
  let dealerTotal = calculateHandValue(dealerHand);
  
  // Player actions
  for (const action of actions) {
    if (action === 'hit') {
      playerHand.push(dealCard());
      playerTotal = calculateHandValue(playerHand);
      
      if (playerTotal > 21) {
        break; // Bust
      }
    }
  }
  
  // Dealer plays (hits until 17+)
  while (dealerTotal < 17 && playerTotal <= 21) {
    dealerHand.push(dealCard());
    dealerTotal = calculateHandValue(dealerHand);
  }
  
  // Determine winner
  let result: 'win' | 'lose' | 'push';
  let payout = 0;
  
  if (playerTotal > 21) {
    result = 'lose';
  } else if (dealerTotal > 21) {
    result = 'win';
    payout = Math.floor(effectiveBet * PAYOUT_MULTIPLIER);
  } else if (playerTotal > dealerTotal) {
    result = 'win';
    payout = Math.floor(effectiveBet * PAYOUT_MULTIPLIER);
  } else if (playerTotal < dealerTotal) {
    result = 'lose';
  } else {
    // In our modified rules, tie = dealer wins
    result = 'lose';
  }
  
  const netProfit = payout - betAmount;
  
  // Transaction
  await db.transaction(async (tx) => {
    // Deduct bet from player
    await tx
      .update(users)
      .set({ money: sql`money - ${betAmount}` })
      .where(eq(users.id, playerId));
    
    // Commission to admin
    await tx
      .insert(adminWallet)
      .values({
        source: 'casino_commission',
        amount: commission,
        description: `Blackjack commission from ${player.username}`
      });
    
    if (result === 'win') {
      // Pay player
      await tx
        .update(users)
        .set({ money: sql`money + ${payout}` })
        .where(eq(users.id, playerId));
      
      // Casino owner loses
      await tx
        .update(users)
        .set({ money: sql`money - ${payout - effectiveBet}` })
        .where(eq(users.id, casino.ownerId));
    } else {
      // Casino owner wins
      await tx
        .update(users)
        .set({ money: sql`money + ${effectiveBet}` })
        .where(eq(users.id, casino.ownerId));
    }
    
    // Record game
    await tx
      .insert(casinoGames)
      .values({
        casinoId,
        playerId,
        gameType: 'blackjack',
        betAmount,
        commission,
        effectiveBet,
        result,
        payout,
        netProfit,
        gameData: {
          playerHand,
          dealerHand,
          playerTotal,
          dealerTotal,
          actions
        }
      });
  });
  
  return {
    result,
    playerHand,
    dealerHand,
    playerTotal,
    dealerTotal,
    payout,
    netProfit
  };
}
```
