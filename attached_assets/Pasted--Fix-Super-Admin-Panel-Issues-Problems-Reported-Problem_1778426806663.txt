export async function PUT(req: Request, { params }: { params: { id: string } }) {
  try {
    const session = await getAdminSession(); // Verify admin auth
    if (!session) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }
    
    const updates = await req.json();
    const playerId = params.id;
    
    // Update player in database
    await db
      .update(players)
      .set(updates)
      .where(eq(players.id, playerId));
    
    return NextResponse.json({ success: true });
  } catch (error) {
    console.error('Update player error:', error);
    return NextResponse.json({ error: 'Failed to update player' }, { status: 500 });
  }
}

Ensure the amount is submitted correctly:

```typescript
async function handleAddMoney(playerId: string, amount: number) {
  try {
    // Validate
    if (amount <= 0) {
      toast.error('Amount must be greater than 0');
      return;
    }
    
    if (isNaN(amount)) {
      toast.error('Please enter a valid number');
      return;
    }
    
    // API call
    const response = await fetch(`/api/admin/players/${playerId}/add-money`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ amount })
    });
    
    if (!response.ok) {
      throw new Error('Failed to add money');
    }
    
    toast.success(`Added $${amount.toLocaleString()} to player`);
    // Refresh player data
    // Close modal
    setAmount(0);
  } catch (error) {
    console.error('Add money error:', error);
    toast.error('Failed to add money');
  }
}
```
