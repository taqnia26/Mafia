### Step 4: Update Rank Badges

Create proper rank badges with styling:

```tsx
// components/RankBadge.tsx
export function RankBadge({ rank }: { rank: number }) {
  const getBadgeStyle = () => {
    if (rank >= 1 && rank <= 3) return 'badge-bronze';
    if (rank >= 4 && rank <= 6) return 'badge-silver';
    if (rank >= 7 && rank <= 9) return 'badge-gold';
    if (rank >= 10 && rank <= 12) return 'badge-platinum';
    return '';
  };
  
  return (
    <div className={`rank-badge ${getBadgeStyle()}`}>
      <span className="rank-number">{rank}</span>
    </div>
  );
}
```
