import { Router } from 'express';
import { db } from '../lib/db';
import { requireAuth } from '../lib/auth';
import { hospitalPatients, players, playerProperties, propertyTypes } from '@workspace/db/schema';
import { eq, and, sql } from 'drizzle-orm';

const router = Router();

// GET /api/hospital/status - حالة المستشفى
router.get('/status', requireAuth, async (req, res) => {
  const player = await db.select().from(players)
    .where(eq(players.clerkId, req.auth.userId))
    .limit(1);

  if (!player[0].isHospitalized) {
    return res.json({ isHospitalized: false });
  }

  const admission = await db.select().from(hospitalPatients)
    .where(and(
      eq(hospitalPatients.playerId, player[0].id),
      eq(hospitalPatients.status, 'admitted')
    ))
    .limit(1);

  res.json({
    isHospitalized: true,
    admission: admission[0],
    hoursLeft: Math.ceil((new Date(admission[0].dischargeAt).getTime() - Date.now()) / (1000 * 60 * 60))
  });
});

// POST /api/hospital/fast-heal - علاج سريع
router.post('/fast-heal', requireAuth, async (req, res) => {
  await db.transaction(async (tx) => {
    const player = await tx.select().from(players)
      .where(eq(players.clerkId, req.auth.userId))
      .limit(1);

    if (!player[0].isHospitalized) {
      throw new Error('Not hospitalized');
    }

    const admission = await tx.select().from(hospitalPatients)
      .where(and(
        eq(hospitalPatients.playerId, player[0].id),
        eq(hospitalPatients.status, 'admitted')
      ))
      .limit(1);

    if (!admission[0]) {
      throw new Error('No active admission');
    }

    const fastHealFee = 50000;

    if (player[0].money < fastHealFee) {
      throw new Error('Insufficient money');
    }

    // خصم المال
    await tx.update(players)
      .set({ money: sql`${players.money} - ${fastHealFee}` })
      .where(eq(players.id, player[0].id));

    // 20% لمالك المستشفى
    const ownerCut = Math.floor(fastHealFee * 0.2);
    await tx.update(players)
      .set({ money: sql`${players.money} + ${ownerCut}` })
      .where(eq(players.id, admission[0].hospitalOwnerId));

    // تسجيل العلاج السريع
    await tx.update(hospitalPatients)
      .set({
        fastHealUsed: true,
        fastHealFeePaid: fastHealFee,
        dischargedAt: new Date(),
        status: 'discharged'
      })
      .where(eq(hospitalPatients.id, admission[0].id));

    // شفاء اللاعب
    await tx.update(players)
      .set({
        health: sql`${players.maxHealth}`,
        isHospitalized: false,
        hospitalizedUntil: null
      })
      .where(eq(players.id, player[0].id));

    res.json({ success: true, healed: true });
  });
});

export default router;

async function dischargePatients() {
  logger.info('Discharging recovered patients...');

  await db.transaction(async (tx) => {
    const recovered = await tx.select()
      .from(hospitalPatients)
      .where(and(
        eq(hospitalPatients.status, 'admitted'),
        sql`${hospitalPatients.dischargeAt} <= NOW()`
      ));

    for (const patient of recovered) {
      // شفاء كامل
      await tx.update(players)
        .set({
          health: sql`${players.maxHealth}`,
          isHospitalized: false,
          hospitalizedUntil: null
        })

      // تحديث السجل
      await tx.update(hospitalPatients)
        .set({
          status: 'discharged',
          dischargedAt: new Date()
        })
        .where(eq(hospitalPatients.id, patient.id));

      // إشعار
      await tx.insert(notifications).values({
        playerId: patient.playerId,

  const gangId = Number(req.params.id);

  await db.transaction(async (tx) => {
    const player = await tx.select().from(players)
      .where(eq(players.clerkId, req.auth.userId))
      .limit(1);

    const gang = await tx.select().from(gangs)
      .where(eq(gangs.id, gangId))
      .limit(1);

    // التحقق من أنه البوس
    if (gang[0].bossId !== player[0].id) {
      throw new Error('Only the boss can appoint royal guards');
    }

    // التحقق من العدد (3 كحد أقصى)
    const currentGuards = await tx.select()
      .from(royalGuards)
      .where(eq(royalGuards.gangId, gangId));

async function payRoyalGuardSalaries() {
  logger.info('Paying royal guard salaries...');

  await db.transaction(async (tx) => {
    const guards = await tx.select()
      .from(royalGuards)
      .where(sql`last_salary_paid_at IS NULL OR last_salary_paid_at < NOW() - INTERVAL '30 days'`);

    for (const guard of guards) {
      const gang = await tx.select().from(gangs)
        .where(eq(gangs.id, guard.gangId))
        .limit(1);

      if (gang[0].treasury >= guard.monthlySalary) {
        // خصم من الخزينة
        await tx.update(gangs)
          .set({ treasury: sql`${gangs.treasury} - ${guard.monthlySalary}` })
          .where(eq(gangs.id, guard.gangId));

        // إضافة للحارس
        await tx.update(players)
          .set({ money: sql`${players.money} + ${guard.monthlySalary}` })
          .where(eq(players.id, guard.guardPlayerId));

        // تحديث السجل
        await tx.update(royalGuards)
          .set({ lastSalaryPaidAt: new Date() })
          .where(eq(royalGuards.id, guard.id));

        logger.info({ guardId: guard.guardPlayerId, salary: guard.monthlySalary }, 'Salary paid');
      } else {
        logger.warn({ gangId: guard.gangId }, 'Insufficient treasury for guard salary');
      }
    }
  });
}

      throw new Error('Only the boss can veto');
    }

    // التحقق من Cooldown (60 يوم)
    if (player[0].lastVetoUsedAt) {
      const daysSince = (Date.now() - new Date(player[0].lastVetoUsedAt).getTime()) / (1000 * 60 * 60 * 24);
      if (daysSince < 60) {
        throw new Error(`Veto cooldown: ${Math.ceil(60 - daysSince)} days remaining`);
      }
    }

    // إلغاء التصويت
    await tx.update(impeachmentVotes)
      .set({ status: 'vetoed' })
      .where(eq(impeachmentVotes.id, voteId));

    await tx.update(players)
      .set({ lastVetoUsedAt: new Date() })
      .where(eq(players.id, player[0].id));

    res.json({ success: true });
  });
});
